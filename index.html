currentMatch = 1; isMatchLocked = false; for (let i = 1; i <= 10; i++) { if(document.getElementById(`s${i}_1`)) document.getElementById(`s${i}_1`).value = 0; if(document.getElementById(`s${i}_2`)) document.getElementById(`s${i}_2`).value = 0; } calculateTotal(); updateNames(); } } window.onload = loadSavedData;
